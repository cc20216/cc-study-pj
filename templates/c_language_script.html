let currentSyntaxIndex = -1; let showingMeaning = false; // 随机抽查功能 const randomCheckBtn = document.getElementById('random-check-btn'); const randomCheckArea = document.getElementById('random-check-area'); const randomSyntax = document.getElementById('random-syntax'); const randomSyntaxMeaning = document.getElementById('random-syntax-meaning'); const showMeaningBtn = document.getElementById('show-meaning-btn'); const nextSyntaxBtn = document.getElementById('next-syntax-btn'); const knownBtn = document.getElementById('known-btn'); if (randomCheckBtn) { randomCheckBtn.addEventListener('click', () => { randomCheckArea.style.display = 'block'; showNextRandomSyntax(); }); } const showNextRandomSyntax = () => { currentSyntaxIndex = Math.floor(Math.random() * syntaxData.length); randomSyntax.textContent = syntaxData[currentSyntaxIndex].syntax; randomSyntaxMeaning.textContent = '点击"显示释义"查看释义'; showingMeaning = false; }; if (showMeaningBtn) { showMeaningBtn.addEventListener('click', () => { if (!showingMeaning) { randomSyntaxMeaning.textContent = syntaxData[currentSyntaxIndex].meaning; showingMeaning = true; } }); } if (nextSyntaxBtn) { nextSyntaxBtn.addEventListener('click', () => { showNextRandomSyntax(); }); } if (knownBtn) { knownBtn.addEventListener('click', () => { // 这里可以添加已掌握语法点的处理逻辑 alert('已标记为掌握！'); showNextRandomSyntax(); }); } // 语法搜索功能 const searchBtn = document.getElementById('search-btn'); const syntaxSearch = document.getElementById('syntax-search'); const searchSyntax = () => { const query = syntaxSearch.value.trim().toLowerCase(); if (!query) { return; } const results = syntaxData.filter(item => item.syntax.toLowerCase().includes(query) || item.meaning.toLowerCase().includes(query) ); if (results.length > 0) { alert(`找到 ${results.length} 个相关语法点：\n${results.map(r => `${r.syntax}: ${r.meaning}`).join('\n')}`); } else { alert('未找到匹配的语法点'); } }; if (searchBtn) { searchBtn.addEventListener('click', searchSyntax); } if (syntaxSearch) { syntaxSearch.addEventListener('keypress', (e) => { if (e.key === 'Enter') { searchSyntax(); } }); }
